DIPLOMACY = {
	c:RUS ?= {
		if = {
			limit = {
				has_diplomatic_pact = {
					who = c:UZH
					type = protectorate
				}
			}
			remove_diplomatic_pact = {
				country = c:UZH
				type = protectorate
			}
		}
	}

	c:PAN ?= {
		if = {
			limit = {
				has_diplomatic_pact = {
					who = c:KAS
					type = vassal
				}
			}
			remove_diplomatic_pact = {
				country = c:KAS
				type = vassal
			}
		}
	}

	c:KAL ?= {
		create_diplomatic_pact = {
			country = c:MAK
			type = vassal
		}
	}
}
